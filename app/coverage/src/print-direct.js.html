// print-direct.js - Standalone print handler with direct DOM manipulation
(function() {
  // Wait for document to be fully loaded
  document.addEventListener('DOMContentLoaded', function() {
    console.log('[print-direct] Initializing print handler');
    
    // Capture the print function reference
    const originalPrint = window.print;
    
    // Override the window.print function
    window.print = function() {
      try {
        console.log('[print-direct] Print requested, preparing content');
        
        // Save the current body HTML
        const originalBodyContent = document.body.innerHTML;
        
        // Find the print view content
        const printViewElement = document.getElementById('gaslands-print-view');
        
        if (printViewElement) {
          console.log('[print-direct] Found print view element, extracting content');
          
          // Extract the print content
          const printContent = printViewElement.innerHTML;
          
          // Replace the entire body with just the print content
          document.body.innerHTML = `
            <div class="print-only-content" style="
              display: block !important; 
              visibility: visible !important;
              position: absolute !important;
              top: 0 !important;
              left: 0 !important;
              width: 100% !important;
              background-color: white !important;
              color: black !important;
              z-index: 9999 !important;
            ">
              ${printContent}
            </div>
          `;
          
          console.log('[print-direct] Content prepared for printing, calling browser print');
          
          // Use the original print function
          originalPrint.call(window);
          
          // Wait for print dialog to close, then restore the original content
          setTimeout(function() {
            console.log('[print-direct] Restoring original page content');
            document.body.innerHTML = originalBodyContent;
            
            // Restore any event handlers or state if needed
            // This is a simple implementation that doesn't preserve event handlers
            console.log('[print-direct] Content restored');
          }, 1000); // Wait 1 second after print dialog closes
        } else {
          console.error('[print-direct] Print view element not found!');
          // Fall back to original print function
          originalPrint.call(window);
        }
      } catch (error) {
        console.error('[print-direct] Error during print preparation:', error);
        // Fall back to original print function
        originalPrint.call(window);
      }
    };
    
    console.log('[print-direct] Print handler initialized');
  });
})();
